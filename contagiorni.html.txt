<div class="count" id="daysCount"></div>

<script>
    const startDate = new Date('2025-03-16');
    function updateCount() {
        const now = new Date();
        const diffTime = now - startDate;
        const diffDays = Math.floor(diffTime / (1000 * 60 * 60 * 24));
        document.getElementById('daysCount').innerText = diffDays + " giorni";
    }
    updateCount();
    setInterval(updateCount, 1000 * 60 * 60); 
</script>
